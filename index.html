从指针赋值到内存回收

1.定义指针： 声明一个 int * 类型的指针变量 p，用于存放整型数据的地址。
int *p; //指针变量名为p，内部可以存储一个地址，地址指向的区域用于存储int类型的数据

2.堆内存申请：使用 malloc 申请堆空间，并将返回的 void * 强制转换为 int *。

p = (int *)malloc(sizeof(int)); //(int *)强制转换为int指针类型（int *）
if (p == NULL) {
    perror("malloc failed");
    exit(-1); // 或者进行其他错误处理
}

3.存入数据： 通过解引用运算符 (*) 访问 p 指向的内存空间。

*p = 100;  //*为解引用运算符，*p可以读取地址p下存储的数据，存入新数据

4. 释放内存：调用 free(p) 释放堆空间。
free(p); //释放指针p指向的堆内存空间，此时 p 变成了“野指针”，依然指向原地址，但该地址已失效。

5.指针指向NULL:将 p 赋值为 NULL
p=NULL；//将指针指向空，确保无法通过p内部存储的地址访问这个地址下的内容

注意事项
1.p 是一个指针变量，内部存储的是地址值。对 p 赋值必须是一个合法地址。*p 代表通过地址访问到的目标内存单元；

2.释放指针p指的是释放地址p下存储的内容，后续还需要将p内存储的地址赋值为空。

LList_t * LList_NewNode(DataType_t data)
{
//1.创建一个新结点并对新结点申请内存
LList_t *New = (LList_t *)calloc(1,sizeof(LList_t));

if (New == NULL)             //错误处理
{
    perror("calloc memory for New is failed");
    exit(1);
}

//2.对新结点的数据域和指针域进行初始化
New->Data = data;
New->next = NULL;

return New;

    phead = Head->next;

bool LList_PosInset(LList_t * Head,int k,DataType_t data)
{
LList_t *Phead = Head;
//1.创建新结点
LList_t *New = LList_NewNode(data);
if (New == NULL)
{
printf(“can not insert new node\n”);
return false;
}

//2.判断链表是否为空，如果为空，则直接插入即可
if (Head == NULL)
{
    Head->next == New;
    return true;
}

//2.寻找到第k个位置
while (Phead != NULL && k > 0)
{
    Phead = Phead->next;
    k--;
}
if (Phead == NULL)
{
    printf("can not insert new node,linkedlist length than k\n");
    return false;
}

//3.插入元素
New->next = Phead->next;
Phead->next = New;

return true;

}

bool LList_Print(LList_t *Head)
{
LList_t *cur = Head->next;
//1.判断链表是否为空，为空则打印失败，返回false
if (LList_Empty(Head))
{
printf(“linkedlist id empty,printing failed!”);
return false;
}

//2.循环打印链表
while (cur != NULL)
{
    printf("%d->\n",cur->Data);
    cur = cur->next;
}

return true;

}

void LList_DelMin(LList_t *Head)
{
//0.如果链表为空或者只有一个结点，退出
if (Head == NULL && Head->next == NULL)
{
printf(“List is empty,deletion failed!\n”);
exit(1);
}

LList_t *min_prev = Head;                   //记录最小值结点的地址
LList_t *min = Head->next;                  //记录最小值结点的地址

LList_t *phead_prev = Head->next;           //记录当前结点的直接前驱地址 
LList_t *phead = phead_prev->next;          //记录当前结点地址


//1.遍历链表，目的是找到最小结点的值
while (phead != NULL)
{
    if(min->Data > phead->Data)
    {
        min_prev = phead_prev;
        min      = phead;
    }
    phead_prev = phead;
    phead = phead->next;
}

//2.删除最小值的结点
min_prev->next = min->next;

//3.释放最小值结点的内存
min->next = NULL;
free(min);

}

bool LList_DelFirstNode(LList_t *Head)
{

{
   printf("LinkedList is Empty!\n");
   return false;
}

//2.删除结点
Head->next == phead->next->next;
phead->next->next = NULL;
free(phead->next);

return true;

}

DataType_t LList_DelK(LList_t *Head,int k)
{
//1.判断链表是否为空
if (LList_Empty(Head))
{
printf(“Deletion failed!”);
return 0;
}
//2.找到倒数第k个位置
LList_t *Pos = Head;
LList_t *Cur = Head;

while(k > 1)
{
    Cur = Cur->next;
    k--;
    if(Cur == NULL)
    {
    printf("The number of linked list nodes is less than k!");
    return 0;
    }
}

while (Cur->next != NULL)
{
    Cur = Cur->next;
    Pos = Pos->next;
}

//3.获取值并打印
printf("The value at the k-th position from the end is:%d\n",Pos->Data);
return 1;

}
